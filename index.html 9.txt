<script>

const shops=document.querySelectorAll(".shop");
const money=[1000,500,200,100,50,20,10,5,2,1,0.5];

const cash=document.getElementById("cash");

const total=document.getElementById("total");

let counts={};

money.forEach(value=>{

counts[value]=0;

cash.innerHTML+=`

<div class="cashRow">

<b>${value>=1?value+" грн":"50 коп"}</b>

<div class="counter">

<button onclick="change(${value},-1)">-</button>

<input

id="m${value}"

type="number"

min="0"

value="0"

oninput="manual(${value})">

<button onclick="change(${value},1)">+</button>

</div>

</div>

`;

});

function change(value,step){

counts[value]=Math.max(0,counts[value]+step);

document.getElementById("m"+value).value=counts[value];

updateTotal();

}

function manual(value){

counts[value]=Number(document.getElementById("m"+value).value)||0;

updateTotal();

}

function updateTotal(){

let sum=0;

money.forEach(value=>{

sum+=value*counts[value];

});

total.innerHTML=sum.toFixed(2)+" грн";

}
shops.forEach(button=>{

button.onclick=()=>{

shops.forEach(item=>item.classList.remove("active"));

button.classList.add("active");

};

});

</script>
